def par_val(**kwargs):
    # Создаем список имен параметров, которым соответствуют строки с более чем двумя словами
    return [name for name, value in kwargs.items() if len(value.split()) > 2]


result = par_val(pp='abba war', fan='oneword', zr='a x')
print(result)
