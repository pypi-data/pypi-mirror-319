def psort(**kwargs):
    # Сортируем ключи по значениям и возвращаем отсортированные имена параметров
    return [key for key, value in sorted(kwargs.items(), key=lambda item: item[1])]


result = psort(c=21, a=22, ac=17, b=16)
print(result)
