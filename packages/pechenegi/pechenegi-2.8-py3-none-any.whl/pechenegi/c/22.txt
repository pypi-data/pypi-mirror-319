def find_max_sum_row(matrix):
    max_sum = 0
    max_row_index = 0
    for i, row in enumerate(matrix):
        row_sum = sum(row)
        if row_sum > max_sum:
            max_sum = row_sum
            max_row_index = i
    return max_row_index, max_sum

matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
]
print(find_max_sum_row(matrix))
